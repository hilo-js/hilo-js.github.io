  2 /**
  3  * @name Graphics
  4  * @class Graphics类包含一组创建矢量图形的方法。
  5  * @augments View
  6  * @module hilo/view/Graphics
  7  * @requires hilo/core/Hilo
  8  * @requires hilo/core/Class
  9  * @property {Number} lineWidth 笔画的线条宽度。默认为1。
 10  * @property {Number} lineAlpha 笔画的线条透明度。默认为1。
 11  * @property {String} lineCap 笔画的线条端部样式。可选值有：butt、round、square等，默认为null。
 12  * @property {String} lineJoin 笔画的线条连接样式。可选值有：miter、round、bevel等，默认为null。
 13  * @property {Number} miterLimit 斜连线长度和线条宽度的最大比率。此属性仅当lineJoin为miter时有效。默认值为10。
 14  * @property {String} strokeStyle 笔画边框的样式。默认值为'0'，即黑色。
 15  * @property {String} fillStyle 内容填充的样式。默认值为'0'，即黑色。
 16  * @property {Number} fillAlpha 内容填充的透明度。默认值为0。
 17  */
 18 var Graphics = (function(){
 19 
 20 var helpContext = document.createElement('canvas').getContext('2d');
 21 
 22 return Class.create(/** @lends Graphics.prototype */{
 23     Extends: View,
 24     /** @constructs */
 25     constructor: function Graphics(properties){
 26         properties = properties || {};
 27         this.id = this.id || properties.id || Hilo.getUid('Graphics');
 28         Graphics.superclass.constructor.call(this, properties);
 29 
 30         this._actions = [];
 31         this._cache = null;
 32     },
 33 
 34     lineWidth: 1,
 35     lineAlpha: 1,
 36     lineCap: null, //'butt', 'round', 'square'
 37     lineJoin: null, //'miter', 'round', 'bevel'
 38     miterLimit: 10,
 39     hasStroke: false,
 40     strokeStyle: '0',
 41     hasFill: false,
 42     fillStyle: '0',
 43     fillAlpha: 0,
 44 
 45     /**
 46      * 指定绘制图形的线条样式。
 47      */
 48     lineStyle: function(thickness, lineColor, alpha, lineCap, lineJoin, miterLimit){
 49         var me = this, addAction = me._addAction;
 50         
 51         addAction.call(me, ['lineWidth', (me.lineWidth = thickness || 1)]);
 52         addAction.call(me, ['strokeStyle', (me.strokeStyle = lineColor || '0')]);
 53         addAction.call(me, ['lineAlpha', (me.lineAlpha = alpha || 1)]);
 54         if(lineCap != undefined) addAction.call(me, ['lineCap', (me.lineCap = lineCap)]);
 55         if(lineJoin != undefined) addAction.call(me, ['lineJoin', (me.lineJoin = lineJoin)]);
 56         if(miterLimit != undefined) addAction.call(me, ['miterLimit', (me.miterLimit = miterLimit)]);
 57         me.hasStroke = true;
 58         return me;
 59     },
 60 
 61     /**
 62      * 指定绘制图形的填充样式和透明度。
 63      */
 64     beginFill: function(fill, alpha){
 65         var me = this, addAction = me._addAction;
 66 
 67         addAction.call(me, ['fillStyle', (me.fillStyle = fill)]);
 68         addAction.call(me, ['fillAlpha', (me.fillAlpha = alpha || 1)]);
 69         me.hasFill = true;
 70         return me;
 71     },
 72 
 73     /**
 74      * 应用并结束笔画的绘制和图形样式的填充。
 75      */
 76     endFill: function(){
 77         var me = this, addAction = me._addAction;
 78 
 79         if(me.hasStroke) addAction.call(me, ['stroke']);
 80         if(me.hasFill) addAction.call(me, ['fill']);
 81         return me;
 82     },
 83 
 84     /**
 85      * 指定绘制图形的线性渐变填充样式。
 86      */
 87     beginLinearGradientFill: function(x0, y0, x1, y1, colors, ratios){
 88         var me = this, gradient = helpContext.createLinearGradient(x0, y0, x1, y1);
 89 
 90         for (var i = 0, len = colors.length; i < len; i++){
 91             gradient.addColorStop(ratios[i], colors[i]);
 92         }
 93         me.hasFill = true;
 94         return me._addAction(['fillStyle', (me.fillStyle = gradient)]);
 95     },
 96 
 97     /**
 98      * 指定绘制图形的放射性渐变填充样式。
 99      */
100     beginRadialGradientFill: function(x0, y0, r0, x1, y1, r1, colors, ratios){
101         var me = this, gradient = helpContext.createRadialGradient(x0, y0, r0, x1, y1, r1);
102         for (var i = 0, len = colors.length; i < len; i++)
103         {
104             gradient.addColorStop(ratios[i], colors[i]);
105         }
106         me.hasFill = true;
107         return me._addAction(['fillStyle', (me.fillStyle = gradient)]);
108     },
109 
110     /**
111      * 开始一个位图填充样式。
112      * @param {HTMLImageElement} image 指定填充的Image对象。
113      * @param {String} repetition 指定填充的重复设置参数。它可以是以下任意一个值：repeat, repeat-x, repeat-y, no-repeat。默认为''。
114      */
115     beginBitmapFill: function(image, repetition){
116         var me = this, pattern = helpContext.createPattern(image, repetition || '');
117         me.hasFill = true;
118         return me._addAction(['fillStyle', (me.fillStyle = pattern)]);
119     },
120 
121     /**
122      * 开始一个新的路径。
123      */
124     beginPath: function(){
125         return this._addAction(['beginPath']);
126     },
127 
128     /**
129      * 关闭当前的路径。
130      */
131     closePath: function(){
132         return this._addAction(['closePath']);
133     },
134 
135     /**
136      * 绘制一个矩形。
137      */
138     drawRect: function(x, y, width, height){
139         return this._addAction(['rect', x, y, width, height]);
140     },
141 
142     /**
143      * 绘制一个复杂的圆角矩形。
144      */
145     drawRoundRectComplex: function(x, y, width, height, cornerTL, cornerTR, cornerBR, cornerBL){
146         var me = this, addAction = me._addAction;
147         addAction.call(me, ['moveTo', x + cornerTL, y]);
148         addAction.call(me, ['lineTo', x + width - cornerTR, y]);
149         addAction.call(me, ['arc', x + width - cornerTR, y + cornerTR, cornerTR, -Math.PI/2, 0, false]);
150         addAction.call(me, ['lineTo', x + width, y + height - cornerBR]);
151         addAction.call(me, ['arc', x + width - cornerBR, y + height - cornerBR, cornerBR, 0, Math.PI/2, false]);
152         addAction.call(me, ['lineTo', x + cornerBL, y + height]);
153         addAction.call(me, ['arc', x + cornerBL, y + height - cornerBL, cornerBL, Math.PI/2, Math.PI, false]);
154         addAction.call(me, ['lineTo', x, y + cornerTL]);
155         addAction.call(me, ['arc', x + cornerTL, y + cornerTL, cornerTL, Math.PI, Math.PI*3/2, false]);
156         return me;
157     },
158 
159     /**
160      * 绘制一个圆角矩形。
161      */
162     drawRoundRect: function(x, y, width, height, cornerSize){
163         return this.drawRoundRectComplex(x, y, width, height, cornerSize, cornerSize, cornerSize, cornerSize);
164     },
165 
166     /**
167      * 绘制一个圆。
168      */
169     drawCircle: function(x, y, radius){
170         return this._addAction(['arc', x + radius, y + radius, radius, 0, Math.PI * 2, 0]);
171     },
172 
173     /**
174      * 绘制一个椭圆。
175      */
176     drawEllipse: function(x, y, width, height){
177         var me = this;
178         if(width == height) return me.drawCircle(x, y, width);
179         
180         var addAction = me._addAction;
181         var w = width / 2, h = height / 2, C = 0.5522847498307933, cx = C * w, cy = C * h;
182         x = x + w;
183         y = y + h;
184         
185         addAction.call(me, ['moveTo', x + w, y]);
186         addAction.call(me, ['bezierCurveTo', x + w, y - cy, x + cx, y - h, x, y - h]);
187         addAction.call(me, ['bezierCurveTo', x - cx, y - h, x - w, y - cy, x - w, y]);
188         addAction.call(me, ['bezierCurveTo', x - w, y + cy, x - cx, y + h, x, y + h]);
189         addAction.call(me, ['bezierCurveTo', x + cx, y + h, x + w, y + cy, x + w, y]);
190         return me;
191     },
192 
193     /**
194      * 根据参数指定的SVG数据绘制一条路径。
195      * 代码示例: 
196      * <p>var path = 'M250 150 L150 350 L350 350 Z';</p>
197      * <p>var shape = new Quark.Graphics({width:500, height:500});</p>
198      * <p>shape.drawSVGPath(path).beginFill('#0ff').endFill();</p>
199      */
200     drawSVGPath: function(pathData){
201         var me = this, addAction = me._addAction,
202             path = pathData.split(/,| (?=[a-zA-Z])/);
203         
204         addAction.call(me, ['beginPath']);
205         for(var i = 0, len = path.length; i < len; i++){
206             var str = path[i], cmd = str[0].toUpperCase(), p = str.substring(1).split(/,| /);
207             if(p[0].length == 0) p.shift();
208 
209             switch(cmd){
210                 case 'M':
211                     addAction.call(me, ['moveTo', p[0], p[1]]);
212                     break;
213                 case 'L':
214                     addAction.call(me, ['lineTo', p[0], p[1]]);
215                     break;
216                 case 'C':
217                     addAction.call(me, ['bezierCurveTo', p[0], p[1], p[2], p[3], p[4], p[5]]);
218                     break;
219                 case 'Z':
220                     addAction.call(me, ['closePath']);
221                     break;
222             }
223         }
224         return me;
225     },

227     /**
228      * 执行全部绘制动作。内部私有方法。
229      * @private
230      */
231     _draw: function(context){
232         var me = this, actions = me._actions, len = actions.length, i;
233         
234         context.beginPath();
235         for(i = 0; i < len; i++){
236             var action = actions[i], 
237                 f = action[0], 
238                 args = action.length > 1 ? action.slice(1) : null;
239             
240             if(typeof(context[f]) == 'function') context[f].apply(context, args);
241             else context[f] = action[1];
242         }
243     },
244 
245     /**
246      * 重写渲染实现。
247      */
248     render: function(renderer, delta){
249         var me = this, canvas = renderer.canvas;
250         if(canvas.getContext){
251             me._draw(renderer.context);
252         }else{
253             var drawable = me.drawable;
254             if(!drawable.image){
255                 drawable.image = me.toImage();
256             }
257             renderer.draw(me);
258         }
259     },
260 
261     /**
262      * 缓存graphics到一个canvas或image。可用来提高渲染效率。
263      */
264     cache: function(toImage){
265         var me = this, cached = me._cache;
266         if(!cached){
267             cached = me._cache = Hilo.createElement('canvas', {width:me.width, height:me.height});
268             me._draw(cached.getContext('2d'));
269         }
270         if(toImage) cached = me._cache = me.toImage();
271 
272         return cached;
273     },
274 
275     /**
276      * 清除缓存。
277      */
278     uncache: function(){
279         this._cache = null;
280     },
281 
282     /**
283      * 把Graphics对象转换成dataURL格式的位图。
284      * @param {String} type 指定转换为DataURL格式的图片mime类型。默认为'image/png'。
285      */
286     toImage: function(type){
287         var me = this, obj = me._cache, w = me.width, h = me.height;
288 
289         if(!obj){
290             obj = Hilo.createElement('canvas', {width:w, height:h});
291             me._draw(obj.getContext('2d'));
292         }
293 
294         if(!(obj instanceof HTMLImageElement)){
295             var src = obj.toDataURL(type || 'image/png');
296             obj = new Image();
297             obj.src = src;
298             obj.width = w;
299             obj.height = h;
300         }
301         
302         return obj;
303     },
304 
305     /**
306      * 清除所有绘制动作并复原所有初始状态。
307      */
308     clear: function(){
309         var me = this;
310 
311         me._actions.length = 0;
312         me._cache = null;
313         
314         me.lineWidth = 1;
315         me.lineAlpha = 1;
316         me.lineCap = null;
317         me.lineJoin = null;
318         me.miterLimit = 10;
319         me.hasStroke = false;
320         me.strokeStyle = '0';
321         me.hasFill = false;
322         me.fillStyle = '0';
323         me.fillAlpha = 1;
324     },
325 
326     /** 
327      * 添加一个绘制动作。内部私有方法。
328      * @private
329      */
330     _addAction: function(action){
331         var me = this;
332         me._actions.push(action);
333         return me;
334     }
