  2 /**
  3  * @class Text类提供简单的文字显示功能。复杂的文本功能可以使用DOMElement。
  4  * @augments View
  5  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。
  6  * @module hilo/view/Text
  7  * @requires hilo/core/Class
  8  * @requires hilo/core/Hilo
  9  * @requires hilo/view/View
 10  * @property {String} text 指定要显示的文本内容。
 11  * @property {String} color 指定使用的字体颜色。
 12  * @property {String} textAlign 指定文本的对齐方式。可以是以下任意一个值：'start', 'end', 'left', 'right', and 'center'。
 13  * @property {Boolean} outline 指定文本是绘制边框还是填充。
 14  * @property {Number} lineSpacing 指定文本的行距。单位为像素。默认值为0。
 15  * @property {Number} textWidth 指示文本内容的宽度，只读属性。仅在canvas模式下有效。
 16  * @property {Number} textHeight 指示文本内容的高度，只读属性。仅在canvas模式下有效。
 17  * @property {String} font 指定使用的字体样式。
 18  */
 19 var Text = Class.create(/** @lends Text.prototype */{
 20     Extends: View,
 21     /** @constructs */
 22     constructor: function Text(properties){
 23         properties = properties || {};
 24         this.id = this.id || properties.id || Hilo.getUid('Text');
 25         Text.superclass.constructor.call(this, properties);
 26 
 27         if(!this.width) this.width = 200; //default width
 28         if(!this.font) this.font = '12px arial'; //default font style
 29     },
 30 
 31     text: null,
 32     color: null,
 33     textAlign: null,
 34     outline: false,
 35     lineSpacing: 0,
 36     textWidth: 0, //read-only
 37     textHeight: 0, //read-only
 38     font: {
 39         get: function(){
 40             return this._font || null;
 41         },
 42         set: function(value){
 43             if(this._font === value) return;
 44             this._font = value;
 45             this._fontHeight = Text.measureFontHeight(value);
 46         }
 47     },
 48 
 49     /**
 50      * 覆盖渲染方法。
 51      */
 52     render: function(renderer, delta){
 53         var me = this, canvas = renderer.canvas;
 54 
 55         if(canvas.getContext){
 56             me._draw(renderer.context);
 57         }else{
 58             var drawable = me.drawable;
 59             var domElement = drawable.domElement;
 60             var style = domElement.style;
 61 
 62             style.font = me.font;
 63             style.textAlign = me.textAlign;
 64             style.color = me.color;
 65             style.width = me.width + 'px';
 66             style.height = me.height + 'px';
 67             style.lineHeight = (me._fontHeight + me.lineSpacing) + 'px';
 68 
 69             domElement.innerHTML = me.text;
 70             renderer.draw(this);
 71         }
 72     },
 73 
 74     /**
 75      * 在指定的渲染上下文上绘制文本。
 76      * @private
 77      */
 78     _draw: function(context){
 79         var me = this, text = me.text.toString();
 80         if(!text) return;
 81 
 82         //set drawing style
 83         context.font = me.font;
 84         context.textAlign = me.textAlign;
 85         context.textBaseline = 'top';
 86         if(me.outline) context.strokeStyle = me.color;
 87         else context.fillStyle = me.color;
 88 
 89         //find and draw all explicit lines
 90         var lines = text.split(/\r\n|\r|\n|<br(?:[ \/])*>/);
 91         var width = 0, height = 0;
 92         var lineHeight = me._fontHeight + me.lineSpacing;
 93         var i, line, w;
 94 
 95         for(i = 0, len = lines.length; i < len; i++){
 96             line = lines[i];
 97             w = context.measureText(line).width;
 98 
 99             //check if the line need to split
100             if(w <= me.width){
101                 me._drawTextLine(context, line, height);
102                 if(width < w) width = w;
103                 height += lineHeight;
104                 continue;
105             }
106 
107             var str = '', oldWidth = 0, newWidth, j, word;
108 
109             for(j = 0, wlen = line.length; j < wlen; j++){
110                 word = line[j];
111                 newWidth = context.measureText(str + word).width;
112 
113                 if(newWidth > me.width){
114                     me._drawTextLine(context, str, height);
115                     if(width < oldWidth) width = oldWidth;
116                     height += lineHeight;
117                     str = word;
118                 }else{
119                     oldWidth = newWidth;
120                     str += word;
121                 }
122 
123                 if(j == wlen - 1){
124                     me._drawTextLine(context, str, height);
125                     if(str !== word && width < newWidth) width = newWidth;
126                     height += lineHeight;
127                 }
128             }
129         }
130 
131         me.textWidth = width;
132         me.textHeight = height;
133         if(!me.height) me.height = height;
134     },
135 
136     /**
137      * 在指定的渲染上下文上绘制一行文本。
138      * @private
139      */
140     _drawTextLine: function(context, text, y){
141         var me = this, x = 0, width = me.width;
142 
143         switch(me.textAlign){
144             case 'center':
145                 x = width * 0.5 >> 0;
146                 break;
147             case 'right':
148             case 'end':
149                 x = width;
150                 break;
151         };
152 
153         if(me.outline) context.strokeText(text, x, y);
154         else context.fillText(text, x, y);
155     },
156 
157     Statics: /** @lends Text */{
158         /**
159          * 测算指定字体样式的行高。
160          * @param {String} font 指定要测算的字体样式。
161          * @return {Number} 返回指定字体的行高。
162          */
163         measureFontHeight: function(font){
164             var docElement = document.documentElement, fontHeight;
165             var elem = Hilo.createElement('div', {style:{font:font, position:'absolute'}, innerHTML:'M'});
166 
167             docElement.appendChild(elem);
168             fontHeight = elem.offsetHeight;
169             docElement.removeChild(elem);
170             return fontHeight;
171         }
172     }
173 
174 });
