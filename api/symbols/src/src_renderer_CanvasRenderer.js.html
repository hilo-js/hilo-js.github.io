  2 /**
  3  * @class canvas画布渲染器。所有可视对象将渲染在canvas画布上。
  4  * @augments Renderer
  5  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。
  6  * @module hilo/renderer/CanvasRenderer
  7  * @requires hilo/core/Class
  8  * @requires hilo/renderer/Renderer
  9  * @property {CanvasRenderingContext2D} context canvas画布的上下文。
 10  */
 11 var CanvasRenderer = Class.create(/** @lends CanvasRenderer.prototype */{
 12     Extends: Renderer,
 13     /** @constructs */
 14     constructor: function CanvasRenderer(properties){
 15         CanvasRenderer.superclass.constructor.call(this, properties);
 16         
 17         var canvas = this.canvas,
 18             width = properties.width,
 19             height = properties.height;
 20             
 21         this.context = canvas.getContext("2d");
 22         if(width && height){
 23             canvas.width = width;
 24             canvas.height = height;
 25         }
 26     },
 27 
 28     context: null,
 29 
 30     /**
 31      * @private
 32      * @see Renderer#startDraw
 33      */
 34     startDraw: function(target){
 35         if(target instanceof Stage){
 36             this.context.clearRect(0, 0, target.width, target.height);
 37         }
 38         this.context.save();
 39     },
 40 
 41     /**
 42      * @private
 43      * @see Renderer#draw
 44      */
 45     draw: function(target){
 46         var drawable = target.drawable;
 47         if(!drawable || !drawable.image) return;
 48         var ctx = this.context, rect = drawable.rect;
 49         ctx.drawImage(drawable.image, rect[0], rect[1], rect[2], rect[3], 0, 0, target.width, target.height);
 50     },
 51 
 52     /**
 53      * @private
 54      * @see Renderer#endDraw
 55      */
 56     endDraw: function(target){
 57         this.context.restore();
 58     },
 59 
 60     /**
 61      * @private
 62      * @see Renderer#transform
 63      */
 64     transform: function(target){
 65         var ctx = this.context,
 66             scaleX = target.scaleX,
 67             scaleY = target.scaleY;
 68         
 69         if(target instanceof Stage){
 70             var style = this.canvas.style,
 71                 oldScaleX = target._scaleX,
 72                 oldScaleY = target._scaleY;
 73 
 74             if((!oldScaleX && scaleX != 1) || (oldScaleX && oldScaleX != scaleX)){
 75                 target._scaleX = scaleX;
 76                 style.width = scaleX * target.width + "px";
 77             }
 78             if((!oldScaleY && scaleY != 1) || (oldScaleY && oldScaleY != scaleY)){
 79                 target._scaleY = scaleY;
 80                 style.height = scaleY * target.height + "px";
 81             }
 82         }else{
 83             var x = target.x,
 84                 y = target.y,
 85                 pivotX = target.pivotX,
 86                 pivotY = target.pivotY,
 87                 rotation = target.rotation % 360;
 88 
 89             if(x != 0 || y != 0) ctx.translate(x, y);
 90             if(rotation != 0) ctx.rotate(rotation * Math.PI / 180);
 91             if(scaleX != 1 || scaleY != 1) ctx.scale(scaleX, scaleY);
 92             if(pivotX != 0 || pivotY != 0) ctx.translate(-pivotX, -pivotY);
 93         }
 94         
 95         if(target.alpha > 0) ctx.globalAlpha *= target.alpha;
 96     },
 97 
 98     /**
 99      * @private
100      * @see Renderer#remove
101      */
102     remove: function(target){
103         if(target instanceof DOMElement){
104             var elem = target.drawable.domElement,
105                 parentElem = elem && elem.parentNode;
106             if(parentElem){
107                 parentElem.removeChild(elem);
108             }
109         }
110     },
111 
112     /**
113      * @private
114      * @see Renderer#clear
115      */
116     clear: function(x, y, width, height){
117         this.context.clearRect(x, y, width, height);
118     }
119 });
