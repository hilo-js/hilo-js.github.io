  2 /**
  3  * @class Container是所有容器类的基类。每个Container都可以添加其他可视对象为子级。
  4  * @augments View
  5  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。
  6  * @module hilo/view/Container
  7  * @requires hilo/core/Hilo
  8  * @requires hilo/core/Class
  9  * @requires hilo/view/View
 10  * @property {Array} children 容器的子元素列表。只读。
 11  * @property {Number} numChildren 容器的子元素数量。只读。
 12  * @property {Boolean} pointerChildren 指示容器的子元素是否能响应用户交互事件。默认为true。
 13  * @property {Boolean} clipChildren 指示是否裁剪超出容器范围的子元素。默认为false。
 14  */
 15 var Container = Class.create(/** @lends Container.prototype */{
 16     Extends: View,
 17     /** @constructs */
 18     constructor: function Container(properties){
 19         properties = properties || {};
 20         this.id = this.id || properties.id || Hilo.getUid("Container");
 21         Container.superclass.constructor.call(this, properties);
 22 
 23         if(this.children) this._updateChildren();
 24         else this.children = [];
 25     },
 26 
 27     children: null,
 28     pointerChildren: true,
 29     clipChildren: false,
 30     numChildren: {
 31         get: function(){
 32             return this.children.length;
 33         }
 34     },
 35 
 36     /**
 37      * 在指定索引位置添加子元素。
 38      * @param {View} child 要添加的子元素。
 39      * @param {Number} index 指定的索引位置，从0开始。
 40      */
 41     addChildAt: function(child, index){
 42         var children = this.children, 
 43             len = children.length, 
 44             parent = child.parent;
 45         
 46         index = index < 0 ? 0 : index > len ? len : index;
 47         var childIndex = this.getChildIndex(child);
 48         if(childIndex == index){
 49             return this;
 50         }else if(childIndex >= 0){
 51             children.splice(childIndex, 1);
 52         }else if(parent){
 53             parent.removeChild(child);
 54         }
 55 
 56         children.splice(index, 0, child);
 57         this._updateChildren(index);
 58 
 59         return this;
 60     },
 61 
 62     /**
 63      * 在最上面添加子元素。
 64      * @param {View} child 要添加的子元素。
 65      */
 66     addChild: function(child){
 67         var total = this.children.length, 
 68             args = arguments;
 69 
 70         for(var i = 0, len = args.length; i < len; i++){
 71             this.addChildAt(args[i], total + i);
 72         }
 73         return this;
 74     },
 75 
 76     /**
 77      * 在指定索引位置删除子元素。
 78      * @param {index} 指定删除元素的索引位置，从0开始。
 79      */
 80     removeChildAt: function(index){
 81         var children = this.children;
 82         if(index < 0 || index >= children.length) return null;
 83 
 84         var child = children[index];
 85         if(child){
 86             var stage = this.stage;
 87             if(stage) stage.renderer.remove(child);
 88             child._parent = null;
 89             child._depth = -1;
 90         }
 91 
 92         children.splice(index, 1);
 93         this._updateChildren(index);
 94 
 95         return child;
 96     },
 97 
 98     /**
 99      * 删除指定的子元素。
100      * @param {View} child 指定要删除的子元素。
101      */
102     removeChild: function(child){
103         return this.removeChildAt(this.getChildIndex(child));
104     },
105 
106     /**
107      * 删除指定id的子元素。
108      * @param {String} 指定要删除的子元素的id。
109      */
110     removeChildById: function(id){
111         var children = this.children, child;
112         for(var i = 0, len = children.length; i < len; i++){
113             child = children[i];
114             if(child.id === id){
115                 this.removeChildAt(i);
116                 return child;
117             }
118         }
119         return null;
120     },
121 
122     /**
123      * 删除所有的子元素。
124      */
125     removeAllChildren: function(){
126         while(this.children.length) this.removeChildAt(0);
127     },
128 
129     /**
130      * 返回指定索引位置的子元素。
131      * @param {Number} index 指定要返回的子元素的索引值，从0开始。
132      */
133     getChildAt: function(index){
134         var children = this.children;
135         if(index < 0 || index >= children.length) return null;
136         return children[index];
137     },
138 
139     /**
140      * 返回指定id的子元素。
141      * @param {String} id 指定要返回的子元素的id。
142      */
143     getChildById: function(id){
144         var children = this.children, child;
145         for(var i = 0, len = children.length; i < len; i++){
146             child = children[i];
147             if(child.id === id) return child;
148         }
149         return null;
150     },
151 
152     /**
153      * 返回指定子元素的索引值。
154      * @param {View} child 指定要返回索引值的子元素。
155      */
156     getChildIndex: function(child){
157         return this.children.indexOf(child);
158     },
159 
160     /**
161      * 设置子元素的索引位置。
162      * @param {View} child 指定要设置的子元素。
163      * @param {Number} index 指定要设置的索引值。
164      */
165     setChildIndex: function(child, index){
166         var children = this.children, 
167             oldIndex = children.indexOf(child);
168 
169         if(oldIndex >= 0 && oldIndex != index){
170             var len = children.length;
171             index = index < 0 ? 0 : index >= len ? len - 1 : index;
172             children.splice(oldIndex, 1);
173             children.splice(index, 0, child);
174             this._updateChildren();
175         }
176         return this;
177     },
178 
179     /**
180      * 交换两个子元素的索引位置。
181      * @param {View} child1 指定要交换的子元素A。
182      * @param {View} child2 指定要交换的子元素B。
183      */
184     swapChildren: function(child1, child2){
185         var children = this.children,
186             index1 = this.getChildIndex(child1), 
187             index2 = this.getChildIndex(child2);
188 
189         child1._depth = index2;
190         children[index2] = child1;
191         child2._depth = index1;
192         children[index1] = child2;
193     },
194 
195     /**
196      * 交换两个指定索引位置的子元素。
197      * @param {Number} index1 指定要交换的索引位置A。
198      * @param {Number} index2 指定要交换的索引位置B。
199      */
200     swapChildrenAt: function(index1, index2){
201         var children = this.children,
202             child1 = this.getChildAt(index1), 
203             child2 = this.getChildAt(index2);
204 
205         child1._depth = index2;
206         children[index2] = child1;
207         child2._depth = index1;
208         children[index1] = child2;
209     },
210 
211     /**
212      * 根据指定键值或函数对子元素进行排序。
213      * @param {Object} keyOrFunction 如果此参数为String时，则根据子元素的某个属性值进行排序；如果此参数为Function时，则根据此函数进行排序。
214      */
215     sortChildren: function(keyOrFunction){
216         var fn = keyOrFunction,
217             children = this.children;
218         if(typeof fn == "string"){
219             var key = fn;
220             fn = function(a, b){
221                 return b[key] - a[key];
222             };
223         }
224         children.sort(fn);
225         this._updateChildren();
226     },
227 
228     /**
229      * 更新子元素。
230      * @private
231      */
232     _updateChildren: function(start, end){
233         var children = this.children, child,
234             start = start || 0,
235             end = end || children.length;
236         for(var i = start; i < end; i++){
237             child = children[i];
238             child._depth = i + 1;
239             child._parent = this;
240         }
241     },
242 
243     /**
244      * 返回是否包含参数指定的子元素。
245      * @param {View} child 指定要测试的子元素。
246      */
247     contains: function(child){
248         return this.getChildIndex(child) >= 0;
249     },
250 
251     /** 
252      * 返回由x和y指定的点下的对象。
253      * @param {Number} x 指定点的x轴坐标。
254      * @param {Number} y 指定点的y轴坐标。
255      * @param {Boolean} usePolyCollision 指定是否使用多边形碰撞检测。默认为false。
256      * @param {Boolean} global 使用此标志表明将查找所有符合的对象，而不仅仅是第一个，即全局匹配。默认为false。
257      * @param {Boolean} eventMode 使用此标志表明将在事件模式下查找对象。默认为false。
258      */
259     getViewAtPoint: function(x, y, usePolyCollision, global, eventMode){
260         var result = global ? [] : null,
261             children = this.children, child, obj;
262         
263         for(var i = children.length - 1; i >= 0; i--){
264             child = children[i];
265             //skip child which is not shown or pointer enabled
266             if(!child || !child.visible || child.alpha <= 0 || (eventMode && !child.pointerEnabled)) continue;
267             //find child recursively
268             if(child.children && child.numChildren && !(eventMode && !child.pointerChildren)){
269                 obj = child.getViewAtPoint(x, y, usePolyCollision, global, eventMode);
270             }
271 
272             if(obj){
273                 if(!global) return obj;
274                 else if(obj.length) result = result.concat(obj);
275             }else if(child.hitTestPoint(x, y, usePolyCollision)){
276                 if(!global) return child;
277                 else result.push(child);
278             }
279         }
280 
281         return global && result.length ? result : null;
282     },
283 
284     /**
285      * 覆盖渲染方法。
286      * @private
287      */
288     render: function(renderer, delta){
289         Container.superclass.render.call(this, renderer, delta);
290         
291         var children = this.children;
292         for(var i = 0, len = children.length; i < len; i++){
293             var child = children[i];
294             child._render(renderer, delta);
295         }
296     }
