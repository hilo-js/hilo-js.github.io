  2 /**
  3  * @name DOMElement
  4  * @class DOMElement是dom元素的包装。
  5  * @augments View
  6  * @module hilo/view/DOMElement
  7  * @requires hilo/core/Hilo
  8  * @requires hilo/core/Class
  9  * @requires hilo/view/View
 10  * @requires hilo/view/Drawable
 11  */
 12 var DOMElement = Class.create(/** @lends DOMElement.prototype */{
 13     Extends: View,
 14     /** @constructs */
 15     constructor: function DOMElement(properties){
 16         properties = properties || {};
 17         this.id = this.id || properties.id || Hilo.getUid("DOMElement");
 18         DOMElement.superclass.constructor.call(this, properties);
 19         
 20         this.drawable = new Drawable();
 21         var elem = this.drawable.domElement = properties.element || Hilo.createElement('div');
 22         elem.id = this.id;
 23     },
 24 
 25     /**
 26      * 覆盖渲染方法。
 27      * @private
 28      */
 29     _render: function(renderer, delta){
 30         var elem = this.drawable.domElement, style = elem.style, parentElem = elem.parentNode;
 31         if(!this.visible || this.alpha <= 0){
 32             style.display = 'none';
 33             return;
 34         }
 35 
 36         style.display = '';
 37         this.fire('beforerender', delta);
 38         Hilo.setElementStyleByView(this);
 39         this.render(renderer, delta);
 40         this.fire('afterrender', delta);
 41     },
 42 
 43     /**
 44      * 覆盖渲染方法。
 45      * @private
 46      */
 47     render: function(renderer, delta){
 48         var canvas = renderer.canvas;
 49         if(canvas.getContext){
 50             var elem = this.drawable.domElement, depth = this.depth, 
 51                 nextElement = canvas.nextSibling, nextDepth;
 52             if(elem.parentNode) return;
 53 
 54             //draw dom element just after stage canvas
 55             while(nextElement && nextElement.nodeType != 3){
 56                 nextDepth = parseInt(nextElement.style.zIndex) || 0;
 57                 if(nextDepth <= 0 || nextDepth > depth){
 58                     break;
 59                 }
 60                 nextElement = nextElement.nextSibling;
 61             }
 62             canvas.parentNode.insertBefore(this.drawable.domElement, nextElement);
 63         }else{
 64             renderer.draw(this);
 65         }
 66     }
 67 });
