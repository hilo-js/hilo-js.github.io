  2 /**
  3  * @name Stage
  4  * @class 舞台是可视对象树的根，可视对象只有添加到舞台或其子对象后才会被渲染出来。
  5  * @augments Container
  6  * @module hilo/view/Stage
  7  * @requires hilo/core/Hilo
  8  * @requires hilo/core/Class
  9  * @requires hilo/view/Container
 10  * @property {HTMLElement} canvas 舞台所对应的画布。它可以是一个canvas或一个普通的div。只读属性。
 11  * @property {Renderer} renderer 舞台渲染器。只读属性。
 12  * @property {Boolean} paused 指示舞台是否暂停刷新渲染。 
 13  * @property {Object} viewport 舞台内容在页面中的渲染区域。包含的属性有：left、top、width、height。只读属性。
 14  */
 15 var Stage = Class.create(/** @lends Stage.prototype */{
 16     Extends: Container,
 17     /** @constructs */
 18     constructor: function Stage(properties){
 19         properties = properties || {};
 20         this.id = this.id || properties.id || Hilo.getUid("Stage");
 21         Stage.superclass.constructor.call(this, properties);
 22         
 23         var canvas = this.canvas;
 24         if(canvas){
 25             if(canvas.getContext){
 26                 this.renderer = new CanvasRenderer(properties);
 27             }else{
 28                 this.renderer = new DOMRenderer(properties);
 29             }
 30         }else{
 31             throw new Error("Stage: canvas is required.");
 32         }
 33         this.updateViewport();
 34     },
 35 
 36     canvas: null,
 37     renderer: null,
 38     paused: false,
 39     viewport: null,
 40 
 41     /**
 42      * 调用tick会触发舞台的更新和渲染。
 43      */
 44     tick: function(delta){
 45         if(!this.paused){
 46             this._render(this.renderer, delta);
 47         }
 48     },
 49 
 50     /**
 51      * 开启/关闭DOM事件功能。
 52      */
 53     enableDOMEvent: function(type, enable){
 54         var me = this,
 55             canvas = me.canvas,
 56             types = typeof type === 'string' ? [type] : type,
 57             handler = me._domListener || (me._domListener = function(e){me._onDOMEvent(e)});
 58 
 59         for(var i = 0; i < types.length; i++){
 60             var type = types[i];
 61             if(enable === false){
 62                 canvas.removeEventListener(type, handler);   
 63             }else{
 64                 canvas.addEventListener(type, handler, false);
 65             }
 66         }
 67     },
 68 
 69     /**
 70      * DOM事件处理函数。此方法会把事件调度到事件的坐标点所对应的可视对象。
 71      * @private
 72      */
 73     _onDOMEvent: function(e){
 74         var type = e.type, event = e, isTouch = type.indexOf('touch') == 0;
 75 
 76         //special for touch event
 77         if(isTouch){
 78             var touches = e.touches, changedTouches = e.changedTouches;
 79             event = (touches && touches.length) ? touches[0] : 
 80                     (changedTouches && changedTouches.length) ? changedTouches[0] : null;
 81             event.type = type;
 82         }
 83 
 84         //calculate stageX/stageY
 85         var x = event.pageX || event.clientX, y = event.pageY || event.clientY, 
 86             viewport = this.viewport;
 87 
 88         event.stageX = x = (x - viewport.left) / this.scaleX;
 89         event.stageY = y = (y - viewport.top) / this.scaleY;
 90 
 91         var obj = this.getViewAtPoint(x, y, true, false, true),
 92             canvas = this.canvas, target = this._eventTarget;
 93 
 94         //fire mouseout/touchout event for last event target
 95         var leave = type === 'mouseout' && !canvas.contains(e.relatedTarget); 
 96         if(target && (target != obj || leave)){
 97             var out = (type === 'touchmove') ? 'touchout' : 
 98                       (type === 'mousemove' || leave || !obj) ? 'mouseout' : null;
 99             if(out) target.fire(out);
100             event.lastEventTarget = target;
101             this._eventTarget = null;
102         }
103         
104         //fire event for current view
105         if(obj && obj.pointerEnabled && type !== 'mouseout'){
106             event.eventTarget = this._eventTarget = obj;
107             obj.fire(event);
108         }
109         
110         //set cursor for current view
111         if(!isTouch){
112             var cursor = (obj && obj.pointerEnabled && obj.useHandCursor) ? 'pointer' : '';
113             canvas.style.cursor = cursor;
114         }
115 
116         //fire event for stage
117         if(leave || type !== "mouseout") this.fire(event);
118     },
119 
120     /**
121      * 更新舞台在页面中的渲染区域。当舞台canvas的样式border、margin、padding等属性更改后，需要调用此方法更新舞台渲染区域。
122      */
123     updateViewport: function(){
124         return this.viewport = Hilo.getElementRect(this.canvas);
125     }
126 
127 });
