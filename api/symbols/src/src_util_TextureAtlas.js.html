  2 /**
  3  * @class TextureAtlas纹理集是将许多小的纹理图片整合到一起的一张大图。这个类可根据一个纹理集数据读取纹理小图、精灵动画等。
  4  * @module hilo/util/TextureAtlas
  5  * @require hilo/core/Class
  6  * @param {Image} image 纹理集图片。
  7  * @param {Array} frameData 纹理集帧数据。每帧的数据格式为：[x, y, width, height]。
  8  * @param {Object} spriteData 纹理集精灵数据。
  9  */
 10 var TextureAtlas = (function(){
 11 
 12 return Class.create({
 13     constructor: function TextureAtlas(image, frameData, spriteData){
 14         var result = parseAtlasData(image, frameData, spriteData);
 15         this._frames = result.frames;
 16         this._sprites = result.sprites;
 17     },
 18 
 19     _frames: null,
 20     _sprites: null,
 21 
 22     /**
 23      * 获取指定索引位置index的帧数据。
 24      */
 25     getFrame: function(index){
 26         var frames = this._frames;
 27         return frames && frames[index];
 28     },
 29 
 30     /**
 31      * 获取指定id的精灵数据。
 32      */
 33     getSprite: function(id){
 34         var sprites = this._sprites;
 35         return sprites && sprites[id];
 36     }
 37 });
 38 
 39 /**
 40  * 解析纹理集帧和精灵数据。
 41  */
 42 function parseAtlasData(image, frameData, spriteData){
 43     if(frameData === undefined){
 44         var data = image;
 45         image = data.image;
 46         frameData = data.frames;
 47         spriteData = data.sprites;
 48     }
 49 
 50     //frames
 51     if(frameData){
 52         var frames = [], frame, obj;
 53 
 54         for(var i = 0, len = frameData.length; i < len; i++){
 55             obj = frameData[i];
 56             frame = {
 57                 image: image,
 58                 rect: [obj[0], obj[1], obj[2], obj[3]],
 59                 pivotX: obj[4] || 0,
 60                 pivotY: obj[5] || 0
 61             };
 62             frames[i] = frame;
 63         }
 64     }
 65 
 66     //sprite frames
 67     if(spriteData){
 68         var sprites = {}, sprite, spriteFrames, spriteFrame;
 69 
 70         for(var s in spriteData){
 71             sprite = spriteData[s];
 72             if(isNumber(sprite)){
 73                 spriteFrames = translateSpriteFrame(frames[sprite]);
 74             }else if(sprite instanceof Array){
 75                 spriteFrames = [];
 76                 for(var i = 0, len = sprite.length; i < len; i++){
 77                     var spriteObj = sprite[i], frameObj;
 78                     if(isNumber(spriteObj)){
 79                         spriteFrame = translateSpriteFrame(frames[spriteObj]);
 80                     }else{
 81                         frameObj = spriteObj.rect;
 82                         if(isNumber(frameObj)) frameObj = frames[spriteObj.rect];
 83                         spriteFrame = translateSpriteFrame(frameObj, spriteObj);
 84                     }
 85                     spriteFrames[i] = spriteFrame;
 86                 }
 87             }
 88             sprites[s] = spriteFrames;
 89         }
 90     }
 91 
 92     return {frames:frames, sprites:sprites};
 93 }
 94 
 95 function isNumber(value){
 96     return typeof value === 'number';
 97 }
 98 
 99 function translateSpriteFrame(frameObj, spriteObj){
100     var spriteFrame = {
101         image: frameObj.image,
102         rect: frameObj.rect,
103         pivotX: frameObj.pivotX || 0,
104         pivotY: frameObj.pivotY || 0
105     };
106 
107     if(spriteObj){
108         spriteFrame.name = spriteObj.name || null;
109         spriteFrame.duration = spriteObj.duration || 0;
110         spriteFrame.stop = !!spriteObj.stop;
111         spriteFrame.next = spriteObj.next || null;
112     }
113 
114     return spriteFrame;
115 }
116 
117 })();
