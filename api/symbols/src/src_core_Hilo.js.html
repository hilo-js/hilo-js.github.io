  7 var Hilo = (function(){
  8 
  9 var win = window, doc = document, docElem = doc.documentElement,
 10     uid = 0;
 11 
 12 return {
 13     /**
 14      * 获取一个全局唯一的id。如Stage1，Bitmap2等。
 15      */
 16     getUid: function(prefix){
 17         var id = ++uid;
 18         if(prefix){
 19             var charCode = prefix.charCodeAt(prefix.length - 1);
 20             if (charCode >= 48 && charCode <= 57) prefix += "_"; // 0至9之间添加下划线
 21             return prefix + id;
 22         }
 23         return id;
 24     },
 25 
 26     /**
 27      * 为指定的可视对象生成一个包含路径的字符串表示形式。如Stage1.Container2.Bitmap3。
 28      */
 29     viewToString: function(view){
 30         var result, obj = view;
 31         while(obj){
 32             if (!obj.id){
 33                 throw new Error("view obj must have id property");
 34             }
 35             result = result ? (obj.id + '.' + result) : obj.id;
 36             obj = obj.parent;
 37         }
 38         return result;
 39     },
 40 
 41     /**
 42      * 简单的浅复制对象。
 43      * @param strict  当为true时只覆盖已有属性
 44      */
 45     copy: function(target, source, strict){ 
 46         for(var key in source){
 47             if(!strict || target.hasOwnProperty(key) || target[key] !== undefined){
 48                 target[key] = source[key];
 49             }
 50         }
 51         return target;
 52     },
 53 
 54     /**
 55      * 浏览器特性。
 56      */
 57     browser: (function(){
 58         var ua = navigator.userAgent;
 59         var data = {
 60             webkit: /webkit/i.test(ua),
 61             ie: /msie/i.test(ua),
 62             chrome: /chrome/i.test(ua),
 63             safari: /safari/i.test(ua),
 64             firefox: /firefox/i.test(ua),
 65             opera: /opera/i.test(ua),
 66             ios: /ios/i.test(ua),
 67             android: /android/i.test(ua),
 68             supportTouch: 'ontouchstart' in win,
 69             supportStorage: 'localStorage' in win,
 70             supportOrientation: 'orientation' in win,
 71             supportDeviceMotion: 'ondevicemotion' in win,
 72             supportCanvas: document.createElement('canvas').getContext != null,
 73         };
 74 
 75         //vendro prefix
 76         var jsVendor = data.jsVendor = data.webkit ? 'webkit' : data.firefox ? 'Moz' : data.opera ? 'O' : data.ie ? 'ms' : '';
 77         var cssVendor = data.cssVendor = '-' + jsVendor + '-';
 78 
 79         //css transform/3d feature dectection
 80         var testElem = doc.createElement('div'), style = testElem.style;
 81         var supportTransform = style[jsVendor + 'Transform'] != undefined;
 82         var supportTransform3D = style[jsVendor + 'Perspective'] != undefined;
 83         if(supportTransform3D){
 84             testElem.id = 'test3d';
 85             style = doc.createElement('style');
 86             style.textContent = '@media ('+ cssVendor +'transform-3d){#test3d{height:3px}}';
 87             doc.head.appendChild(style);
 88 
 89             docElem.appendChild(testElem);
 90             supportTransform3D = testElem.offsetHeight == 3;
 91             doc.head.removeChild(style);
 92             docElem.removeChild(testElem);
 93         };
 94         data.supportTransform = supportTransform;
 95         data.supportTransform3D = supportTransform3D;
 96 
 97         return data;
 98     })(),

100     /**
101      * 获取DOM元素在页面中的内容显示区域。
102      */
103     getElementRect: function(elem){
104         try{
105             //this fails if it's a disconnected DOM node
106             var bounds = elem.getBoundingClientRect();
107         }catch(e){
108             bounds = {top: elem.offsetTop, left: elem.offsetLeft, width:elem.offsetWidth, height:elem.offsetHeight};
109         }
110 
111         var offsetX = (win.pageXOffset || docElem.scrollLeft) - (docElem.clientLeft || 0);
112         var offsetY = (win.pageYOffset || docElem.scrollTop) - (docElem.clientTop || 0);
113         var styles = win.getComputedStyle ? getComputedStyle(elem) : elem.currentStyle;
114         var parseIntFn = parseInt;
115         var padLeft = parseIntFn(styles.paddingLeft) + parseIntFn(styles.borderLeftWidth);
116         var padTop = parseIntFn(styles.paddingTop) + parseIntFn(styles.borderTopWidth);
117         var padRight = parseIntFn(styles.paddingRight) + parseIntFn(styles.borderRightWidth);
118         var padBottom = parseIntFn(styles.paddingBottom) + parseIntFn(styles.borderBottomWidth);
119 
120         return {
121             left: bounds.left + offsetX + padLeft,
122             top: bounds.top + offsetY + padTop,
123             width: bounds.right - padRight - bounds.left - padLeft,
124             height: bounds.bottom - padBottom - bounds.top - padTop
125         };
126     },
127 
128     /**
129      * 创建一个DOM元素。可指定属性和样式。
130      */
131     createElement: function(type, properties){
132         var elem = doc.createElement(type), p, val, s;
133         for(p in properties){
134             val = properties[p];
135             if(p === 'style'){
136                 for(s in val) elem.style[s] = val[s];
137             }else{
138                 elem[p] = val;
139             }
140         }
141         return elem;
142     },
143 
144     /**
145      * 设置可视对象DOM元素的CSS样式。
146      */
147     setElementStyleByView: function(obj){
148         var drawable = obj.drawable,
149             image = drawable.image,
150             rect = drawable.rect,
151             style = drawable.domElement.style,
152             prefix = Hilo.browser.jsVendor, px = 'px';
153         
154         style.pointerEvents = 'none';
155         style.display = (!obj.visible || obj.alpha <= 0) ? 'none' : '';
156         style.width = obj.width + px;
157         style.height = obj.height + px;
158         style.opacity = obj.alpha;
159         style.zIndex = obj.depth;
160         style[prefix + 'TransformOrigin'] = obj.pivotX + px + ' ' + obj.pivotY + px;
161         style[prefix + 'Transform'] = this.getTransformCSS(obj);
162 
163         if(image){
164             if(style.backgroundImage.indexOf(image.src) != 4){
165                 style.backgroundImage = 'url(' + image.src + ')';
166             }
167             if(rect){
168                 style.backgroundPosition = -rect[0] + px + ' ' + -rect[1] + px;
169             }
170         }
171         
172         if(obj.mask){
173             style[prefix + 'MaskImage'] = obj.mask.drawable.domElement.style.backgroundImage;
174             style[prefix + 'MaskRepeat'] = 'no-repeat';
175             style[prefix + 'MaskPosition'] = obj.mask.x + px + ' ' + obj.mask.y + px;
176         }
177     },
178 
179     /**
180      * 生成可视对象的CSS变换样式。
181      */
182     getTransformCSS: function(obj){
183         var use3d = this.browser.supportTransform3D, 
184             str3d = use3d ? '3d' : '';
185 
186         return 'translate' + str3d + '(' + (obj.x - obj.pivotX) + 'px, ' + (obj.y - obj.pivotY) + (use3d ? 'px, 0px)' : 'px)')
187              + 'rotate' + str3d + (use3d ? '(0, 0, 1, ' : '(') + obj.rotation + 'deg)'
188              + 'scale' + str3d + '(' + obj.scaleX + ', ' + obj.scaleY + (use3d ? ', 1)' : ')');
189     }
190 };
