  2 /**
  3  * @private
  4  * @class javascript或JSONP加载器。
  5  * @module hilo/loader/ScriptLoader
  6  * @requires hilo/core/Class
  7  */
  8 var ScriptLoader = Class.create({
  9     load: function(data){
 10         var me = this, src = data.src, isJSONP = data.type == 'jsonp';
 11 
 12         if(isJSONP){
 13             var callbackName = data.callbackName || 'callback';
 14             var callback = data.callback || 'jsonp' + (++ScriptLoader._count);
 15             var win = window;
 16 
 17             if(!win[callback]){
 18                 win[callback] = function(result){
 19                     delete win[callback];
 20                 }
 21             }
 22         }
 23 
 24         if(isJSONP) src += (src.indexOf('?') == -1 ? '?' : '&') + callbackName + '=' + callback;
 25         if(data.noCache) src += (src.indexOf('?') == -1 ? '?' : '&') + 't=' + (+new Date());
 26 
 27         var script = document.createElement('script');
 28         script.type = 'text/javascript';
 29         script.async = true;
 30         script.onload = me.onLoad.bind(me);
 31         script.onerror = me.onError.bind(me);
 32         script.src = src;
 33         if(data.id) script.id = data.id;
 34         document.getElementsByTagName('head')[0].appendChild(script);
 35     },
 36 
 37     onLoad: function(e){
 38         var script = e.target;
 39         script.onload = script.onerror = null;
 40         return script;
 41     },
 42 
 43     onError: function(e){
 44         var script = e.target;
 45         script.onload = script.onerror = null;
 46         return e;
 47     },
 48 
 49     Statics: {
 50         _count: 0
 51     }
 52 
 53 });
