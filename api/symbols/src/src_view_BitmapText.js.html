  2 /**
  3  * @class BitmapText类提供使用位图文本的功能。当前仅支持单行文本。
  4  * @augments Container
  5  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。
  6  * @module hilo/view/BitmapText
  7  * @requires hilo/core/Class
  8  * @requires hilo/core/Hilo
  9  * @requires hilo/view/Container
 10  * @property {Object} glyphs 位图字体的字形集合。格式为：{letter:{image:img, rect:[0,0,100,100]}}。
 11  * @property {Number} letterSpacing 字距，即字符间的间隔。默认值为0。
 12  */
 13 var BitmapText = Class.create(/** @lends BitmapText.prototype */{
 14     Extends: Container,
 15     constructor: function BitmapText(properties){
 16         properties = properties || {};
 17         this.id = this.id || properties.id || Hilo.getUid('BitmapText');
 18         BitmapText.superclass.constructor.call(this, properties);
 19 
 20         this.pointerChildren = false; //disable user events for single letters
 21     },
 22 
 23     glyphs: null,
 24     letterSpacing: 0,
 25     text: {
 26         get: function(){
 27             return this._text || '';
 28         },
 29         set: function(str){
 30             var me = this, str = str.toString(), len = str.length;
 31             if(me._text == str) return;
 32             me._text = str;
 33             me.removeAllChildren();
 34 
 35             var i, charStr, charGlyph, charObj, width = 0, height = 0;
 36 
 37             for(i = 0; i < len; i++){
 38                 charStr = str.charAt(i);
 39                 charGlyph = me.glyphs[charStr];
 40                 if(charGlyph){
 41                     charObj = new Bitmap({
 42                         image: charGlyph.image,
 43                         rect: charGlyph.rect,
 44                         x: width
 45                     });
 46                     width += (width > 0 ? me.letterSpacing : 0) + charGlyph.rect[2];
 47                     heigth = Math.max(height, charGlyph.rect[3]);
 48                     me.addChild(charObj);
 49                 }
 50             }
 51 
 52             me.width = width;
 53             me.height = height;
 54         }
 55     },
 56 
 57     /**
 58      * 返回能否使用当前指定的字体显示提供的字符串。
 59      * @param {String} str 要检测的字符串。
 60      * @returns {Boolean} 是否能使用指定字体。
 61      */
 62     hasGlyphs: function(str){
 63         var glyphs = this.glyphs;
 64         if(!glyphs) return false;
 65 
 66         var str = str.toString(), len = str.length, i;
 67         for(i = 0; i < len; i++){
 68             if(!glyphs[str.charAt(i)]) return false;
 69         }
 70         return true;
 71     }
 72 
 73 });
