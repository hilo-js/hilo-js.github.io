  2 /**
  3  * @name Matrix
  4  * @class Matrix类表示一个转换矩阵，它确定如何将点从一个坐标空间映射到另一个坐标空间。
  5  * @module hilo/geom/Matrix
  6  * @requires hilo/core/Class
  7  */
  8 var Matrix = Class.create(/** @lends Matrix.prototype */{
  9     /** @constructs */
 10     constructor: function Matrix(a, b, c, d, tx, ty){
 11         this.a = a;
 12         this.b = b;
 13         this.c = c;
 14         this.d = d;
 15         this.tx = tx;
 16         this.ty = ty;
 17     },
 18     
 19     concat: function(mtx){
 20         var args = arguments, 
 21             a = this.a, b = this.b, c = this.c, d = this.d, 
 22             tx = this.tx, ty = this.ty;
 23 
 24         if(args.length >= 6){
 25             var ma = args[0], mb = args[1], mc = args[2],
 26                 md = args[3], mx = args[4], my = args[5];
 27         }else{
 28             ma = mtx.a;
 29             mb = mtx.b;
 30             mc = mtx.c;
 31             md = mtx.d;
 32             mx = mtx.tx;
 33             my = mtx.ty;
 34         }
 35         
 36         this.a = a * ma + b * mc;
 37         this.b = a * mb + b * md;
 38         this.c = c * ma + d * mc;
 39         this.d = c * mb + d * md;
 40         this.tx = tx * ma + ty * mc + mx;
 41         this.ty = tx * mb + ty * md + my;
 42         return this;
 43     },
 44 
 45     rotate: function(angle){
 46         var sin = Math.sin(angle), cos = Math.cos(angle),
 47             a = this.a, b = this.b, c = this.c, d = this.d,
 48             tx = this.tx, ty = this.ty;
 49         
 50         this.a = a * cos - b * sin;
 51         this.b = a * sin + b * cos;
 52         this.c = c * cos - d * sin;
 53         this.d = c * sin + d * cos;
 54         this.tx = tx * cos - ty * sin;
 55         this.ty = tx * sin + ty * cos;
 56         return this;
 57     },
 58 
 59     scale: function(sx, sy){
 60         this.a *= sx;
 61         this.d *= sy;
 62         this.tx *= sx;
 63         this.ty *= sy;
 64         return this;
 65     },
 66 
 67     translate: function(dx, dy){
 68         this.tx += dx;
 69         this.ty += dy;
 70         return this;
 71     },
 72 
 73     identity: function(){
 74         this.a = this.d = 1;
 75         this.b = this.c = this.tx = this.ty = 0;
 76         return this;
 77     },
 78 
 79     invert: function(){
 80         var a = this.a;
 81         var b = this.b;
 82         var c = this.c;
 83         var d = this.d;
 84         var tx = this.tx;
 85         var i = a * d - b * c;
 86         
 87         this.a = d / i;
 88         this.b = -b / i;
 89         this.c = -c / i;
 90         this.d = a / i;
 91         this.tx = (c * this.ty - d * tx) / i;
 92         this.ty = -(a * this.ty - b * tx) / i;
 93         return this;
 94     },
 95 
 96     transformPoint: function(point, round, returnNew){
 97         var x = point.x * this.a + point.y * this.c + this.tx,
 98             y = point.x * this.b + point.y * this.d + this.ty;
 99 
100         if(round){
101             x = x + 0.5 >> 0;
102             y = y + 0.5 >> 0;
103         }
104         if(returnNew) return {x:x, y:y};
105         point.x = x;
106         point.y = y;
107         return point;
108     }
109 
110 });
