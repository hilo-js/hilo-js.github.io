  2 /**
  3  * @private
  4  * @class 图片资源加载器。
  5  * @module hilo/loader/ImageLoader
  6  * @requires hilo/core/Class
  7  */
  8 var ImageLoader = Class.create({
  9     load: function(data){
 10         var me = this;
 11 
 12         var image = new Image();
 13         image.onload = me.onLoad.bind(me);
 14         image.onerror = image.onabort = me.onError.bind(me);
 15         image.src = data.src + (data.noCache ? (data.src.indexOf('?') == -1 ? '?' : '&') + 't=' + (+new Date) : '');
 16     },
 17 
 18     onLoad: function(e){
 19         var image = e.target;
 20         image.onload = image.onerror = image.onabort = null;
 21         return image;
 22     },
 23 
 24     onError: function(e){
 25         var image = e.target;
 26         image.onload = image.onerror = image.onabort = null;
 27         return e;
 28     }
 29 
 30 });
