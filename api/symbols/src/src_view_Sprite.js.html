  2 /**
  3  * @class 精灵类。
  4  * @augments View
  5  * @module hilo/view/Sprite
  6  * @requires hilo/core/Hilo
  7  * @requires hilo/core/Class
  8  * @requires hilo/view/View
  9  * @param properties 指定精灵的各种属性。
 10  * @property {number} currentFrame 当前播放帧的索引。从0开始。
 11  * @property {number} numFrames 精灵动画的总帧数。
 12  * @property {boolean} paused 判断精灵是否暂停。默认为false。
 13  * @property {boolean} loop 判断精灵是否可以循环播放。默认为true。
 14  * @property {boolean} timeBased 指定精灵动画是否是以时间为基准。默认为false，即以帧为基准。
 15  * @property {number} interval 精灵动画的帧间隔。如果timeBased为true，则单位为毫秒，否则为帧数。
 16  */
 17 var Sprite = Class.create(/** @lends Sprite.prototype */{
 18     Extends: View,
 19     /** @constructs */
 20     constructor: function Sprite(properties){
 21         properties = properties || {};
 22         this.id = this.id || properties.id || Hilo.getUid("Sprite");
 23         Sprite.superclass.constructor.call(this, properties);
 24 
 25         this._frames = [];
 26         this._frameNames = {};
 27         this.drawable = new Drawable();
 28         if(properties.frames) this.addFrame(properties.frames);
 29     },
 30 
 31     _frames: null, //所有帧的集合
 32     _frameNames: null, //带名字name的帧的集合
 33     _frameElapsed: 0, //当前帧持续的时间或帧数
 34     _currentFrame: 0, //当前帧的索引
 35 
 36     paused: false,
 37     loop: true,
 38     timeBased: false,
 39     interval: 1,
 40     currentFrame: {
 41         get: function(){
 42             return this._currentFrame;
 43         }
 44     },
 45     numFrames: {
 46         get: function(){
 47             return this._frames ? this._frames.length : 0;
 48         }
 49     },
 50 
 51     addFrame: function(frame){
 52         var total = this._frames.length;
 53         if(frame instanceof Array){
 54             for(var i = 0, len = frame.length; i < len; i++){
 55                 this.setFrame(frame[i], total + i);
 56             }
 57         }else{
 58             this.setFrame(frame, total);
 59         }
 60         return this;
 61     },
 62 
 63     setFrame: function(frame, index){
 64         var frames = this._frames,
 65             total = frames.length;
 66         index = index < 0 ? 0 : index > total ? total : index;
 67         frames[index] = frame;
 68         if(frame.name) this._frameNames[frame.name] = frame;
 69         if(index == 0 && !this.width || !this.height){
 70             this.width = frame.rect[2];
 71             this.height = frame.rect[3];
 72         }
 73     },
 74 
 75     getFrame: function(indexOrName){
 76         if(typeof indexOrName === 'number'){
 77             var frames = this._frames;
 78             if(indexOrName < 0 || indexOrName >= frames.length) return null;
 79             return frames[indexOrName];
 80         }
 81         return this._frameNames[indexOrName];
 82     },
 83 
 84     getFrameIndex: function(frameValue){
 85         var frames = this._frames,
 86             total = frames.length,
 87             index = -1;
 88         if(typeof frameValue === 'number'){   
 89             index = frameValue;
 90         }else{
 91             var frame = typeof frameValue === 'string' ? this._frameNames[frameValue] : frameValue;
 92             if(frame){
 93                 for(var i = 0; i < total; i++){
 94                     if(frame === frames[i]){
 95                         index = i;
 96                         break;
 97                     }
 98                 }
 99             }
100         }
101         return index;
102     },
103 
104     play: function(){
105         this.paused = false;
106     },
107 
108     stop: function(){
109         this.paused = true;
110     },
111 
112     goto: function(indexOrName, pause){
113         var total = this._frames.length,
114             index = this.getFrameIndex(indexOrName);
115 
116         this._currentFrame = index < 0 ? 0 : index >= total ? total - 1 : index;
117         this.paused = pause;
118     },
119 
120     _render: function(renderer, delta){
121         var frameIndex = this._nextFrame(delta),
122             frame = this._frames[frameIndex];
123 
124         this.drawable.init(frame);
125         Sprite.superclass._render.call(this, renderer, delta);
126     },
127 
128     _nextFrame: function(delta){
129         var frames = this._frames,
130             total = frames.length,
131             frameIndex = this._currentFrame, 
132             frame = frames[frameIndex],
133             duration = frame.duration || this.interval,
134             elapsed = this._frameElapsed;
135 
136         //calculate the current frame elapsed frames/time
137         var value = (frameIndex == 0 && !this.drawable) ? 0 : elapsed + (this.timeBased ? delta : 1);
138         elapsed = this._frameElapsed = value < duration ? value : 0;
139 
140         if(frame.stop || !this.loop && frameIndex >= total - 1){
141             this.stop();
142         }
143 
144         if(!this.paused && elapsed == 0){
145             if(frame.next != null){
146                 //jump to the specified frame
147                 frameIndex = this.getFrameIndex(frame.next);
148             }else if(frameIndex >= total - 1){
149                 //at the end of the frames, go back to first frame
150                 frameIndex = 0;
151             }else if(this.drawable){
152                 //normal go forward to next frame
153                 frameIndex++;
154             }
155             
156             this._currentFrame = frameIndex;
157         }
158 
159         this.fire('enterframe', {frame:frameIndex});
160         
161         return frameIndex;
162     }
163 });
