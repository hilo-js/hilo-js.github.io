  2 /**
  3  * @class DOM+CSS3渲染器。将可视对象以DOM元素方式渲染出来。
  4  * @augments Renderer
  5  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。
  6  * @module hilo/renderer/DOMRenderer
  7  * @requires hilo/core/Class
  8  * @requires hilo/core/Hilo
  9  * @requires hilo/renderer/Renderer
 10  */
 11 var DOMRenderer = (function(){
 12 
 13 return Class.create({
 14     Extends: Renderer,
 15     constructor: function DOMRenderer(properties){
 16         DOMRenderer.superclass.constructor.call(this, properties);
 17     },
 18 
 19     /**
 20      * @private
 21      * @see Renderer#startDraw
 22      */
 23     startDraw: function(target){
 24         //prepare drawable
 25         var drawable = (target.drawable = target.drawable || new Drawable());
 26         drawable.domElement = drawable.domElement || createDOMDrawable(target, drawable);
 27     },
 28 
 29     /**
 30      * @private
 31      * @see Renderer#draw
 32      */
 33     draw: function(target){
 34         var parent = target.parent,
 35             targetElem = target.drawable.domElement,
 36             currentParent = targetElem.parentNode;
 37 
 38         if(parent){
 39             var parentElem = parent.drawable.domElement;
 40             if(parentElem != currentParent){
 41                 parentElem.appendChild(targetElem);
 42             }
 43         }else if(target instanceof Stage && !currentParent){
 44             this.canvas.appendChild(targetElem);
 45         }
 46     },
 47 
 48     /**
 49      * @private
 50      * @see Renderer#transform
 51      */
 52     transform: function(target){
 53         Hilo.setElementStyleByView(target);
 54     },
 55 
 56     /**
 57      * @private
 58      * @see Renderer#remove
 59      */
 60     remove: function(target){
 61         var targetElem = target.drawable.domElement,
 62             parentElem = targetElem.parentNode;
 63 
 64         if(parentElem) parentElem.removeChild(targetElem);
 65     },
 66 
 67     /**
 68      * @private
 69      * @see Renderer#hide
 70      */
 71     hide: function(target){
 72         var elem = target.drawable && target.drawable.domElement;
 73         if(elem) elem.style.display = "none";
 74     }
 75 });
 76 
 77 /**
 78  * 创建一个可渲染的DOM，可指定tagName，如canvas或div。
 79  * @param {Object} view 一个可视对象或类似的对象。
 80  * @param {Object} imageObj 指定渲染的image及相关设置，如绘制区域rect。
 81  * @return {HTMLElement} 新创建的DOM对象。
 82  * @private
 83  */
 84 function createDOMDrawable(view, imageObj){
 85     var tag = view.tagName || "div",
 86         img = imageObj.image,
 87         w = view.width || (img && img.width),
 88         h =  view.height || (img && img.height),
 89         elem = Hilo.createElement(tag), style = elem.style;
 90 
 91     if(view.id) elem.id = view.id;
 92     style.position = "absolute";
 93     style.left = (view.left || 0) + "px";
 94     style.top = (view.top || 0) + "px";
 95     style.width = w + "px";
 96     style.height = h + "px";
 97 
 98     if(tag == "canvas"){
 99         elem.width = w;
100         elem.height = h;
101         if(img){
102             var ctx = elem.getContext("2d");
103             var rect = imageObj.rect || [0, 0, w, h];       
104             ctx.drawImage(img, rect[0], rect[1], rect[2], rect[3], 
105                          (view.x || 0), (view.y || 0), 
106                          (view.width || rect[2]), 
107                          (view.height || rect[3]));
108         }
109     }else{
110         style.opacity = view.alpha != undefined ? view.alpha : 1;
111         if(view instanceof Stage || view.clipChildren) style.overflow = "hidden";
112         if(img && img.src){
113             style.backgroundImage = "url(" + img.src + ")";
114             var bgX = view.rectX || 0, bgY = view.rectY || 0;
115             style.backgroundPosition = (-bgX) + "px " + (-bgY) + "px";
116         }
117     }
118     return elem;
119 }
120 
121 })();
