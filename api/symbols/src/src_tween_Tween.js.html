  2 /**
  3  * @class Tween类提供缓动功能。
  4  * @module hilo/util/Tween
  5  * @requires hilo/core/Class
  6  * @property {Object} target 缓动目标。只读属性。
  7  * @property {Number} time 缓动总时长。单位毫秒。
  8  * @property {Number} delay 缓动延迟时间。单位毫秒。
  9  * @property {Boolean} paused 缓动是否暂停。默认为false。
 10  * @property {Boolean} loop 缓动是否循环。默认为false。
 11  * @property {Boolean} reverse 缓动是否反转播放。默认为false。
 12  * @property {Number} interval 缓动间隔。默认为0。
 13  * @property {Function} ease 缓动变化函数。默认为null。
 14  * @property {Tween} next 下一个缓动变化对象。默认为null。
 15  * @property {Function} onUpdate 缓动更新回调函数。默认为null。
 16  * @property {Function} onComplete 缓动结束回调函数。默认为null。
 17  */
 18 var Tween = (function(){
 19 
 20 return Class.create({
 21     constructor: function Tween(target, newProps, params){
 22         var me = this;
 23         me.target = target;
 24 
 25         me._oldProps = {};
 26         me._newProps = {};
 27         me._deltaProps = {};
 28         me._startTime = 0;
 29         me._lastTime = 0;
 30         me._pausedTime = 0;
 31         me._pausedStartTime = 0;
 32         me._reverseFlag = 1;
 33         me._frameTotal = 0;
 34         me._frameCount = 0;
 35 
 36         for(var p in newProps){       
 37             var oldVal = target[p], newVal = newProps[p];
 38             if(oldVal !== undefined){
 39                 if(typeof oldVal == 'number' && typeof newVal == 'number'){
 40                     me._oldProps[p] = oldVal;
 41                     me._newProps[p] = newVal;
 42                     me._deltaProps[p] = newVal - oldVal;
 43                 }
 44             }
 45         }
 46 
 47         for(var p in params){
 48             me[p] = params[p];
 49         }
 50     },
 51 
 52     target: null,
 53     time: 0,
 54     delay: 0,
 55     paused: false,
 56     loop: false,
 57     reverse: false,
 58     interval: 0,
 59     ease: null,
 60     next: null,
 61 
 62     onUpdate: null,
 63     onComplete: null,
 64 
 65     /**
 66      * 设置缓动对象的初始和目标属性。
 67      * @param oldProps 缓动对象的初始属性。
 68      * @param newProps 缓动对象的目标属性。
 69      */
 70     setProps: function(oldProps, newProps){
 71         for(var p in oldProps){
 72             this.target[p] = this._oldProps[p] = oldProps[p];
 73         }
 74         for(var p in newProps){
 75             this._newProps[p] = newProps[p];
 76             this._deltaProps[p] = newProps[p] - this.target[p];
 77         }
 78     },
 79 
 80     /**
 81      * 初始化Tween类。
 82      * @private
 83      */
 84     _init: function(){
 85         this._startTime = Date.now() + this.delay;
 86         this._pausedTime = 0;
 87         if(this.interval > 0) this._frameTotal = Math.round(this.time / this.interval);
 88         Tween.add(this);
 89     },
 90 
 91     /**
 92      * 启动缓动动画的播放。
 93      */
 94     start: function(){   
 95         this._init();
 96         this.paused = false;
 97     },
 98 
 99     /**
100      * 停止缓动动画的播放。
101      */
102     stop: function(){
103         Tween.remove(this);
104     },
105 
106     /**
107      * 暂停缓动动画的播放。
108      */
109     pause: function(){   
110         this.paused = true;
111         this._pausedStartTime = Date.now();
112     },
113 
114     /**
115      * 恢复缓动动画的播放。
116      */
117     resume: function(){   
118         this.paused = false;
119         this._pausedTime += Date.now() - this._pausedStartTime;
120     },
121 
122     /**
123      * Tween类的内部更新方法。
124      * @private
125      */
126     _update: function(){
127         if(this.paused) return;
128         var now = Date.now();
129         var elapsed = now - this._startTime - this._pausedTime;
130         if(elapsed < 0) return;
131         
132         this._lastTime = now;
133         
134         var ratio = this._frameTotal > 0 ? (++this._frameCount / this._frameTotal) : (elapsed / this.time);
135         if(ratio > 1) ratio = 1;
136         var value = this.ease ? this.ease(ratio) : ratio;
137 
138         for(var p in this._oldProps){
139             this.target[p] = this._oldProps[p] + this._deltaProps[p] * this._reverseFlag * value;
140         }
141         
142         if(this.onUpdate) this.onUpdate(value);
143 
144         if(ratio >= 1){   
145             if(this.reverse){
146                 var tmp = this._oldProps;
147                 this._oldProps = this._newProps;
148                 this._newProps = tmp;
149                 this._startTime = Date.now();
150                 this._frameCount = 0;
151                 this._reverseFlag *= -1;
152             }else if(this.loop){
153                 for(var p in this._oldProps) this.target[p] = this._oldProps[p];
154                 this._startTime = Date.now();
155                 this._frameCount = 0;
156             }else{
157                 Tween.remove(this);
158                 var next = this.next, nextTween;
159                 if(next){
160                     if(next instanceof Tween){
161                         nextTween = next;
162                         next = null;
163                     }else{
164                         nextTween = next.shift();
165                     }
166                     if(nextTween){
167                         nextTween.next = next;
168                         nextTween.start();
169                     }
170                 }
171             }
172 
173             if(this.onComplete) this.onComplete(this);
174             if(this.reverse && !this.loop) this.reverse = false;
175         }
176     },
177 
178     Statics: {
179         _tweens: [],
180 
181         /**
182          * 更新所有Tween实例。
183          * @static
184          */
185         tick: function(){
186             var tweens = this._tweens, i = tweens.length;
187             while(--i >= 0) tweens[i]._update();
188         },
189 
190         /**
191          * 添加Tween实例。
192          * @static
193          */
194         add: function(tween){
195             if(this._tweens.indexOf(tween) == -1) this._tweens.push(tween);
196             return this;
197         },
198 
199         /**
200          * 删除Tween实例。
201          * @staitc
202          */
203         remove: function(tween){
204             if(tween){
205                 var tweens = this._tweens;
206                 var index = tweens.indexOf(tween);
207                 if(index > -1) tweens.splice(index, 1);
208             }
209             return this;
210         },
211 
212         /**
213          * 删除所有Tween实例。
214          * @staitc
215          */
216         removeAll: function(){
217             this._tweens.length = 0;
218         },
219 
220         /**
221          * 创建一个缓动动画，让目标对象从当前属性变换到目标属性。
222          * @param target 缓动目标对象
223          * @param toProps 缓动目标对象的目标属性。
224          * @param params 缓动动画的参数。
225          */
226         to: function(target, toProps, params){
227             var tween = new Tween(target, toProps, params);
228             tween._init();
229             return tween;
230         },
231 
232         /**
233          * 创建一个缓动动画，让目标对象从指定的起始属性变换到当前属性。
234          * @param target 缓动目标对象
235          * @param toProps 缓动目标对象的起始属性。
236          * @param params 缓动动画的参数。
237          */
238         from: function(target, fromProps, params){
239             var tween = new Tween(target, fromProps, params);
240             var tmp = tween._oldProps;
241             tween._oldProps = tween._newProps;
242             tween._newProps = tmp;
243             tween._reverseFlag = -1;
244 
245             for(var p in tween._oldProps) target[p] = tween._oldProps[p];
246 
247             tween._init();
248             return tween;
249         }
250     }
251 
252 });
253 
254 })();
