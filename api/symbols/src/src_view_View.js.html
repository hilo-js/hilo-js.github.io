  2 /**
  3  * @class View类是所有可视对象或组件的基类。
  4  * @param {Object} properties 创建对象的属性参数。可包含此类所有可写属性。
  5  * @module hilo/view/View
  6  * @requires hilo/core/Hilo
  7  * @requires hilo/core/Class
  8  * @requires hilo/event/EventMixin
  9  * @property {String} id 可视对象的唯一标识符。
 10  * @property {Number} x 可视对象的x轴坐标。默认值为0。
 11  * @property {Number} y 可视对象的y轴坐标。默认值为0。
 12  * @property {Number} width 可视对象的宽度。默认值为0。
 13  * @property {Number} height 可视对象的高度。默认值为0。
 14  * @property {Number} alpha 可视对象的透明度。默认值为1。
 15  * @property {Number} rotation 可视对象的旋转角度。默认值为0。
 16  * @property {Boolean} visible 可视对象是否可见。默认为可见，即true。
 17  * @property {Number} pivotX 可视对象的中心点的x轴坐标。默认值为0。
 18  * @property {Number} pivotY 可视对象的中心点的y轴坐标。默认值为0。
 19  * @property {Number} scaleX 可视对象在x轴上的缩放比例。默认为不缩放，即1。
 20  * @property {Number} scaleY 可视对象在y轴上的缩放比例。默认为不缩放，即1。
 21  * @property {Boolean} pointerEnabled 可视对象是否接受交互事件。默认为接受交互事件，即true。
 22  * @property {Container} parent 可视对象的父容器。只读属性。
 23  * @property {Stage} stage 可视对象的舞台实例。只读属性。
 24  * @property {Number} depth 可视对象的深度，也即z轴的序号。只读属性。
 25  * @property {Number} measuredWidth 可视对象的测量宽度。只读属性。
 26  * @property {Number} measuredHeight 可视对象的测量高度。只读属性。
 27  * @property {Drawable} drawable 可视对象的可绘制对象。
 28  * @property {Array} boundsArea 可视对象的区域顶点数组。格式为：[{x:10, y:10}, {x:20, y:20}]。
 29  */
 30 var View = (function(){
 31 
 32 return Class.create(/** @lends View.prototype */{
 33     Mixes: EventMixin,
 34     constructor: function View(properties){
 35         properties = properties || {};
 36         this.id = this.id || properties.id || Hilo.getUid("View");
 37         Hilo.copy(this, properties, true);
 38     },
 39 
 40     id: null,
 41     x: 0,
 42     y: 0,
 43     width: 0,
 44     height: 0,
 45     alpha: 1,
 46     rotation: 0,
 47     visible: true,
 48     pivotX: 0,
 49     pivotY: 0,
 50     scaleX: 1,
 51     scaleY: 1,
 52     pointerEnabled: true,
 53     drawable: null,
 54     boundsArea: null,
 55     parent: {
 56         get: function(){
 57             return this._parent || null;
 58         }
 59     },
 60     stage: {
 61         get: function(){
 62             var obj = this, parent;
 63             while(parent = obj.parent) obj = parent;
 64             if(obj instanceof Stage) return obj;
 65             return null;
 66         }
 67     },
 68     depth: {
 69         get: function(){
 70             return this._parent && this._depth || -1;
 71         }
 72     },
 73     measuredWidth: {
 74         get: function(){
 75             return this.width * this.scaleX;
 76         }   
 77     },
 78     measuredHeight: {
 79         get: function(){
 80             return this.height * this.scaleY;
 81         }
 82     },
 83 
 84     /**
 85      * 从父容器里删除此对象。
 86      */
 87     removeFromParent: function(){
 88         var parent = this.parent;
 89         if(parent) parent.removeChild(this);
 90     },
 91 
 92     /**
 93      * 获取可视对象在舞台全局坐标系内的外接矩形以及所有顶点坐标。
 94      * @returns {Array} 可视对象的顶点坐标数组vertexs。另vertexs还包含属性：
 95      * <ul>
 96      * <li><b>x</b> - 可视对象的外接矩形x轴坐标。</li>
 97      * <li><b>y</b> - 可视对象的外接矩形y轴坐标。</li>
 98      * <li><b>width</b> - 可视对象的外接矩形的宽度。</li>
 99      * <li><b>height</b> - 可视对象的外接矩形的高度。</li>
100      * </ul>
101      */
102     getBounds: function(){
103         var w = this.width, h = this.height,
104             mtx = this.getConcatenatedMatrix(),
105             poly = this.boundsArea || [{x:0, y:0}, {x:w, y:0}, {x:w, y:h}, {x:0, y:h}],
106             vertexs = [], point, x, y, minX, maxX, minY, maxY;
107         
108         for(var i = 0, len = poly.length; i < len; i++){
109             point = mtx.transformPoint(poly[i], true, true);
110             x = point.x;
111             y = point.y;
112 
113             if(i == 0){
114                 minX = maxX = x;
115                 minY = maxY = y;
116             }else{
117                 if(minX > x) minX = x;
118                 else if(maxX < x) maxX = x;
119                 if(minY > y) minY = y;
120                 else if(maxY < y) maxY = y;
121             }
122             vertexs[i] = point;
123         }
124         
125         vertexs.x = minX;
126         vertexs.y = minY;
127         vertexs.width = maxX - minX;
128         vertexs.height = maxY - minY;
129         return vertexs;
130     },
131 
132     /**
133      * 获取可视对象相对于其某个祖先（默认为舞台Stage）的连接矩阵。
134      * @param {View} ancestor 可视对象的相对的祖先容器。
135      * @private
136      */
137     getConcatenatedMatrix: function(ancestor){
138         var mtx = new Matrix(1, 0, 0, 1, 0, 0);
139 
140         if(ancestor !== this){
141             for(var o = this; o.parent && o.parent != ancestor; o = o.parent){       
142                 var cos = 1, sin = 0, 
143                     rotation = o.rotation % 360,
144                     pivotX = o.pivotX, pivotY = o.pivotY, 
145                     scaleX = o.scaleX, scaleY = o.scaleY;
146                 
147                 if(rotation){
148                     var r = rotation * Math.PI / 180;
149                     cos = Math.cos(r);
150                     sin = Math.sin(r);
151                 }
152                 
153                 if(pivotX != 0) mtx.tx -= pivotX;
154                 if(pivotY != 0) mtx.ty -= pivotY;
155                 mtx.concat(cos*scaleX, sin*scaleX, -sin*scaleY, cos*scaleY, o.x, o.y);
156             }
157         }
158         return mtx;
159     },
160 
161     /**
162      * 检测由x和y参数指定的点是否在其外接矩形之内。
163      * @param {Number} x 要检测的点的x轴坐标。
164      * @param {Number} y 要检测的点的y轴坐标。
165      * @param {Boolean} usePolyCollision 是否使用多边形碰撞检测。默认为false。
166      * @returns {Boolean} 点是否在可视对象之内。
167      */
168     hitTestPoint: function(x, y, usePolyCollision){
169         var bound = this.getBounds(),
170             hit = x >= bound.x && x <= bound.x + bound.width && 
171                   y >= bound.y && y <= bound.y + bound.height;
172         
173         if(hit && usePolyCollision){
174             hit = pointInPolygon(x, y, bound);
175         }
176         return hit;
177     },
178 
179     /**
180      * 检测object参数指定的对象是否与其相交。
181      * @param {View} object 要检测的可视对象。
182      * @param {Boolean} usePolyCollision 是否使用多边形碰撞检测。默认为false。
183      */
184     hitTestObject: function(object, usePolyCollision){
185         var b1 = this.getBounds(), 
186             b2 = object.getBounds(),
187             hit = b1.x <= b2.x + b2.width && b2.x <= b1.x + b1.width && 
188                   b1.y <= b2.y + b2.height && b2.y <= b1.y + b1.height;
189         
190         if(hit && usePolyCollision){
191             hit = polygonCollision(b1, b2);
192         }
193         return !!hit;
194     },
195 
196     /**
197      * 可视对象的基本渲染实现，用于框架内部或高级开发使用。通常应该重写render方法。
198      * @param {Renderer} renderer 渲染器。
199      * @param {Number} delta 渲染时时间偏移量。
200      * @protected
201      */
202     _render: function(renderer, delta){
203         renderer = this.renderer || renderer;
204         if(!this.visible || this.alpha <= 0){
205             renderer.hide(this);
206             return;
207         }
208         
209         this.fire('beforerender', delta);
210         renderer.startDraw(this);
211         renderer.transform(this);
212         this.render(renderer, delta);
213         renderer.endDraw(this);
214         this.fire('afterrender', delta);
215     },
216 
217     /**
218      * 可视对象的具体渲染逻辑。子类可通过覆盖此方法实现自己的渲染。
219      * @param {Renderer} renderer 渲染器。
220      * @param {Number} delta 渲染时时间偏移量。
221      */
222     render: function(renderer, delta){
223         renderer.draw(this);
224     },
225 
226     /**
227      * 返回可视对象的字符串表示。
228      * @returns {String} 可视对象的字符串表示。
229      */
230     toString: function(){
231         return Hilo.viewToString(this);
232     }

235 /**
236  * @private
237  */
238 function pointInPolygon(x, y, poly){
239     var cross = 0, onBorder = false, minX, maxX, minY, maxY;        
240     
241     for(var i = 0, len = poly.length; i < len; i++){
242         var p1 = poly[i], p2 = poly[(i+1)%len];           
243         
244         if(p1.y == p2.y && y == p1.y){
245             p1.x > p2.x ? (minX = p2.x, maxX = p1.x) : (minX = p1.x, maxX = p2.x);
246             if(x >= minX && x <= maxX){
247                 onBorder = true;
248                 continue;
249             }
250         }
251         
252         p1.y > p2.y ? (minY = p2.y, maxY = p1.y) : (minY = p1.y, maxY = p2.y);
253         if(y < minY || y > maxY) continue;
254         
255         var nx = (y - p1.y)*(p2.x - p1.x) / (p2.y - p1.y) + p1.x;
256         if(nx > x) cross++;
257         else if(nx == x) onBorder = true;        
258     }
259     
260     return onBorder || (cross % 2 == 1);
261 }

263 /**
264  * @private
265  */
266 function polygonCollision(poly1, poly2){   
267     var result = doSATCheck(poly1, poly2, {overlap:-Infinity, normal:{x:0, y:0}});
268     if(result) return doSATCheck(poly2, poly1, result);
269     return false;
270 }

272 /**
273  * @private
274  */
275 function doSATCheck(poly1, poly2, result){
276     var len1 = poly1.length, len2 = poly2.length, 
277         currentPoint, nextPoint, distance, 
278         min1, max1, min2, max2, dot, overlap, normal = {x:0, y:0};
279     
280     for(var i = 0; i < len1; i++){
281         currentPoint = poly1[i];
282         nextPoint = poly1[(i < len1-1 ? i+1 : 0)];
283         
284         normal.x = currentPoint.y - nextPoint.y;
285         normal.y = nextPoint.x - currentPoint.x;
286         
287         distance = Math.sqrt(normal.x * normal.x + normal.y * normal.y);
288         normal.x /= distance;
289         normal.y /= distance;
290         
291         min1 = max1 = poly1[0].x * normal.x + poly1[0].y * normal.y;        
292         for(var j = 1; j < len1; j++){
293             dot = poly1[j].x * normal.x + poly1[j].y * normal.y;
294             if(dot > max1) max1 = dot;
295             else if(dot < min1) min1 = dot;
296         }
297         
298         min2 = max2 = poly2[0].x * normal.x + poly2[0].y * normal.y;        
299         for(j = 1; j < len2; j++){
300             dot = poly2[j].x * normal.x + poly2[j].y * normal.y;
301             if(dot > max2) max2 = dot;
302             else if(dot < min2) min2 = dot;
303         }
304         
305         if(min1 < min2){
306             overlap = min2 - max1;
307             normal.x = -normal.x;
308             normal.y = -normal.y;
309         }else{
310             overlap = min1 - max2;
311         }
312         
313         if(overlap >= 0){
314             return false;
315         }else if(overlap > result.overlap){
316             result.overlap = overlap;
317             result.normal.x = normal.x;
318             result.normal.y = normal.y;
319         }
320     }
321     
322     return result;
323 }
